class Solution {
    void zigZag(int arr[], int n) 
    {
    for(int i=0;i<n-1;i++)
    {
    if(i%2==0)
    {
    if(arr[i+1]<arr[i])
    {
      int  temp=arr[i];
        arr[i]=arr[i+1];
        arr[i+1]=temp;
    }
    }
    else
    {
     if(arr[i+1]>arr[i])
    {
      int  temp=arr[i];
        arr[i]=arr[i+1];
        arr[i+1]=temp;
    }   
    }
    }
      
    }

}
